# Problem

# Solution
```cpp
class Solution {
public:
    int atMost(vector<int>& nums, int goal){
    int i=0,j=0,sum=0,out=0, N=nums.size();

        while(j<N){
            out+=nums[j++];
            for(;(out>goal)&&(i<j);i++){
                out-=nums[i];
            }
            sum+=j-i;
        }

        return sum;
    }
    int numSubarraysWithSum(vector<int>& nums, int goal) {
        return atMost(nums, goal) - atMost(nums, goal-1);
    }
};
```
